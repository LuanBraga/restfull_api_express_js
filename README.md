# restfull_api_express_js

packages = nodemon, body-parser and consign.

start api => npm start

GET => /api/v1/products

POST => /api/v1/products/ + req body {
    "code": "",
    "name": "",
    "unitPrice": "",
    "description": "",
    "fabricationDate": "",
    "expirationDate": "",
    "inStock": ""
}

DELETE AND PUT => /api/v1/products/:productId
